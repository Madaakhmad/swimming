            'events' => Event::query()
                ->select([
                    'events.*',
                    'users.nama_lengkap AS nama_author',
                    'categories.nama_kategori'
                ])
                ->join('users', 'users.uid', '=', 'events.uid_author')
                ->join('categories', 'categories.uid', '=', 'events.uid_kategori')
                ->limit(1)
                ->all(),
            'categories' => Category::query()->select(['uid', 'nama_kategori'])->all(),
            'users' => User::query()->select(['uid', 'nama_lengkap'])->all(),
        ], 200);
